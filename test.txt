def test_confidence_scoring_flash(df, theme_col, theme_definitions, model_name):
    model = GenerativeModel(model_name)

    for idx in range(2):  # Test only first 2 rows
        row = df.iloc[idx]
        prompt = generate_flash_prompt(row, theme_col, theme_definitions)

        try:
            response = model.generate_content(
                prompt,
                generation_config={
                    "temperature": 0.3,
                    "max_output_tokens": 512
                }
            )
            print(f"\n=== Row {idx + 1} ===")
            print("Prompt:\n", prompt)
            print("Response:\n", response.text)

        except Exception as e:
            print(f"Error in Gemini response for row {idx + 1}: {e}")
