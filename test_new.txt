def evaluate_theme_ai(pro_pct_AI, has_sub_themes_AI, pro_confidence_AI, is_first_column_AI, validity_label_AI):
    """
    Evaluates the theme classification using your specific dataset columns.
    
    Args:
        pro_pct_AI (float): The numeric score (0.0 to 1.0).
        has_sub_themes_AI (int): 1 if subthemes exist, 0 if empty.
        pro_confidence_AI (str): Judge confidence (e.g., 'High').
        is_first_column_AI (int/bool): 1 (True) if first theme, 0 (False) otherwise.
        validity_label_AI (str): Combined label (e.g., 'Valid-High', 'Invalid-Medium').
    """
    
    # --- Logic Block 1: Initial Check ---
    # Condition: if score = 0 OR (subthemes is empty AND score = 1)
    
    if (pro_pct_AI == 0) or (has_sub_themes_AI == 0 and pro_pct_AI == 1):
        if pro_confidence_AI == "High":
            return "EUC-MC REVIEW", pro_pct_AI
        else:
            return "UNRELATED", 0

    # --- Logic Block 2: If it IS the First Theme ---
    elif is_first_column_AI:
        if validity_label_AI == "Valid-High" and pro_pct_AI > 0.5:
            return "CORE", pro_pct_AI
        elif validity_label_AI == "Valid-High" and pro_pct_AI <= 0.5:
            return "EUC-HC", pro_pct_AI
        elif validity_label_AI == "Invalid-High" or validity_label_AI == "Valid-Medium":
            return "EUC-MC", pro_pct_AI
        else:
            return "NA", pro_pct_AI

    # --- Logic Block 3: If it is NOT the First Theme (and score != 0) ---
    elif (not is_first_column_AI) and (pro_pct_AI != 0):
        if validity_label_AI == "Valid-High":
            return "EUC-HC", pro_pct_AI
        elif validity_label_AI == "Valid-Medium":
            return "EUC-MC", pro_pct_AI
        elif validity_label_AI == "Valid-Low":
            return "UNRELATED", pro_pct_AI
            
        # Checking combined Invalid labels
        elif validity_label_AI == "Invalid-Medium":
            return "EUC-MC", pro_pct_AI
        elif validity_label_AI == "Invalid-Low":
            return "LOOSELY", pro_pct_AI
        else:
            return "NA", pro_pct_AI

    # --- Logic Block 4: Fallback ---
    else:
        return "UNRELATED (REJECT REVIEW)", 0
